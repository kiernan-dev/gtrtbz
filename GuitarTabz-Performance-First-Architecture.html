class PerformanceOptimizedDetector {
  async detectChord(audioBuffer: AudioBuffer): Promise<ChordResult> {
    const audioQuality = this.analyzeAudioQuality(audioBuffer);
    const complexity = this.estimateComplexity(audioBuffer);
    
    // Route based on performance requirements
    if (complexity.isBasic && audioQuality.isClean) {
      // 18ms average - perfect for real-time
      return await this.tonalDetection(audioBuffer);
    }
    
    if (complexity.isAdvanced || audioQuality.isNoisy) {
      // 90ms average - still acceptable for accuracy
      return await this.moisesDetection(audioBuffer);
    }
    
    // Parallel processing for ambiguous cases
    const [local, api] = await Promise.allSettled([
      this.tonalDetection(audioBuffer),
      this.moisesDetection(audioBuffer)
    ]);
    
    // Return whichever completes first with acceptable confidence
    return this.selectBestResult(local, api);
  }
}

class SmartChordProcessor {
  // Tier 1: Instant feedback (15-25ms)
  async instantFeedback(audio: AudioBuffer) {
    const quickResult = await TonalJS.quickDetect(audio);
    if (quickResult.confidence > 0.9) {
      this.displayFeedback(quickResult); // Green light instantly
      this.enhanceInBackground(audio); // Validate with API async
    }
  }
  
  // Tier 2: Accurate analysis (80-120ms)  
  async accurateAnalysis(audio: AudioBuffer) {
    const [moises, chordai] = await Promise.allSettled([
      MoisesAPI.analyze(audio),
      ChordAI.analyze(audio)
    ]);
    
    return this.consolidateResults(moises, chordai);
  }
  
  // Tier 3: Deep insights (background processing)
  async deepInsights(audio: AudioBuffer) {
    const theory = TonalJS.getTheoryInsights(audio);
    const fingering = TonicTS.getOptimalFingering(audio);
    const alternatives = await this.getAlternativeVoicings(audio);
    
    return { theory, fingering, alternatives };
  }
}
